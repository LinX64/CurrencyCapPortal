Usage Examples

Latest Rates

const rates = await fetch('https://linx64.github.io/CurrencyCapPortal/latest.json')
  .then(res => res.json());

Period-Based History

// Get last week
const weekData = await fetch('https://linx64.github.io/CurrencyCapPortal/history/1w.json')
  .then(res => res.json());

// Get last month for charts
const monthData = await fetch('https://linx64.github.io/CurrencyCapPortal/history/1m.json')
  .then(res => res.json());

// Get last year for trends
const yearData = await fetch('https://linx64.github.io/CurrencyCapPortal/history/1y.json')
  .then(res => res.json());

// Get all historical data (5 years)
const allData = await fetch('https://linx64.github.io/CurrencyCapPortal/history/5y.json')
  .then(res => res.json());

// Available periods: 1d, 1w, 1m, 1y, 5y

AI Predictions

// Get short-term predictions (7 days)
const predictions = await fetch('https://linx64.github.io/CurrencyCapPortal/predictions/short.json')
  .then(res => res.json());

// Find USD predictions
const usdPrediction = predictions.predictions.find(p => p.currencyCode === 'usd');

console.log('Currency:', usdPrediction.prediction.currencyName);
console.log('Current Price:', usdPrediction.prediction.currentPrice.buy);
console.log('Trend:', usdPrediction.prediction.trend); // BULLISH, BEARISH, NEUTRAL, VOLATILE
console.log('Confidence:', usdPrediction.prediction.confidenceScore); // 0.5 to 0.95

// Tomorrow's prediction
const tomorrow = usdPrediction.prediction.predictions[0];
console.log('Tomorrow:', tomorrow.date);
console.log('Predicted Buy:', tomorrow.predictedBuy);
console.log('Range:', tomorrow.lowerBound, '-', tomorrow.upperBound);

Get Prediction Metadata

// Check what predictions are available
const meta = await fetch('https://linx64.github.io/CurrencyCapPortal/predictions/index.json')
  .then(res => res.json());

console.log('Available:', meta.availableConfigurations);
console.log('Total currencies:', meta.stats.totalCurrencies);
console.log('Success rate:', meta.stats.successfulPredictions);
